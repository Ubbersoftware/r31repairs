# 31Repairs Phase 0 — Foundation & Design System Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Stand up the 31Repairs Next.js app with the 31-repairs-design system, Firebase auth + roles, a security-rules baseline, a landing page shell, and a static service catalog rendered from the PRD seed pricing.

**Architecture:** Next.js 15 App Router (TypeScript, npm) hosted on Vercel; Firebase (`lmsb2b`, namespaced `r31_`) for Auth/Firestore/Storage. Styling is the `31-repairs-design` skill's `tokens.css` (verbatim) plus its component recipes implemented as React components with CSS Modules — **no Tailwind**. Auth is a client-side context (the real security boundary is Firestore rules + Auth custom claims). Rules are tested against the Firestore emulator.

**Tech Stack:** Next.js 15, React 19, TypeScript (strict), Firebase Web SDK v10+, firebase-admin (bootstrap script), Vitest + React Testing Library, @firebase/rules-unit-testing, firebase-tools (emulator for rules tests), Inter via `next/font`.

## Global Constraints

- **Design authority:** the `31-repairs-design` skill governs all color/type/spacing/radius/shadow/motion. Use the token, never a raw hex. Accent is electric blue `#2962FF`. Dark is default (`:root`); light is `<html data-theme="light">`.
- **No Tailwind, no CSS framework.** `src/styles/tokens.css` is the skill's file verbatim. Component styles are CSS Modules reading only tokens.
- **One primary (filled-blue) button per view.** No new brand colors/gradients/fonts. No pure `#000`/`#fff` large surfaces. Must render clean at 360px with no horizontal scroll (except intentional table scroll). Tap targets ≥ 44px. Honor `prefers-reduced-motion`.
- **Money** is stored/handled in **thebe (integer minor units)**; display via `formatPula`. Currency is BWP, shown as `P`. English only.
- **Firestore collections** are namespaced `r31_*`; **Storage paths** are rooted under `r31/`.
- **Firebase config** comes only from environment variables. `NEXT_PUBLIC_*` for web config; service-account JSON is server-only and never committed.
- **TDD:** red → green → refactor. Commit after each task's tests pass. Conventional Commit messages.
- **Spec:** [`docs/specs/2026-06-21-architecture-design.md`](2026-06-21-architecture-design.md). **PRD:** [`docs/31Repairs-PRD.md`](../31Repairs-PRD.md).

`src/lib/theme.ts`:
```ts
export type Theme = 'dark' | 'light'
export const THEME_KEY = 'r31-theme'

export function getInitialTheme(): Theme {
  if (typeof window === 'undefined') return 'dark'
  const stored = localStorage.getItem(THEME_KEY)
  if (stored === 'light' || stored === 'dark') return stored
  return window.matchMedia?.('(prefers-color-scheme: light)').matches ? 'light' : 'dark'
}

export function applyTheme(theme: Theme): void {
  const root = document.documentElement
  if (theme === 'light') root.setAttribute('data-theme', 'light')
  else root.removeAttribute('data-theme')
  localStorage.setItem(THEME_KEY, theme)
}
```

```tsx
'use client'
import { useEffect, useState } from 'react'
import { applyTheme, getInitialTheme, type Theme } from '@/lib/theme'
import styles from './ThemeToggle.module.css'

export function ThemeToggle() {
  const [theme, setTheme] = useState<Theme>('dark')
  useEffect(() => { setTheme(getInitialTheme()) }, [])
  function toggle() {
    const next: Theme = theme === 'dark' ? 'light' : 'dark'
    applyTheme(next); setTheme(next)
  }
  return (
    <button type="button" className={styles.toggle} onClick={toggle}
      aria-label={`Switch theme (currently ${theme})`}>
      {theme === 'dark' ? '☀' : '☾'}
    </button>
  )
}
```

```tsx
import Link from 'next/link'
import styles from './Button.module.css'
import type { ComponentProps } from 'react'

type Variant = 'primary' | 'secondary' | 'ghost'
type Common = { variant?: Variant; size?: 'md' | 'lg'; block?: boolean }

export function Button(
  props: Common & (({ href: string } & ComponentProps<typeof Link>) | ({ href?: undefined } & ComponentProps<'button'>))
) {
  const { variant = 'primary', size = 'md', block, className, ...rest } = props as Common & { className?: string; href?: string }
  const cls = [styles.btn, styles[variant], size === 'lg' && styles.lg, block && styles.block, className]
    .filter(Boolean).join(' ')
  if ('href' in props && props.href) return <Link {...(rest as ComponentProps<typeof Link>)} className={cls} />
  return <button {...(rest as ComponentProps<'button'>)} className={cls} />
}
```

```ts
export function toThebe(pula: number): number { return Math.round(pula * 100) }

export function formatPula(thebe: number): string {
```

```ts
import { SEED_PRICES } from './seed'

export function priceFor(serviceSlug: string, modelId: string, variant: string | null = null): number | null {
  const cell = SEED_PRICES.find(p =>
    p.serviceSlug === serviceSlug && p.modelId === modelId &&
    (p.variant ?? null) === (variant ?? null) && p.available)
  return cell ? cell.amount : null
}

export function fromPrice(serviceSlug: string): number | null {
  const cells = SEED_PRICES.filter(p => p.serviceSlug === serviceSlug && p.available)
  return cells.length ? Math.min(...cells.map(c => c.amount)) : null
}
```

```ts
import { doc, getDoc, serverTimestamp, setDoc } from 'firebase/firestore'
import { auth, db } from './client'

export function newUserDoc(u: { uid: string; email: string | null; displayName: string | null }) {
  return { role: 'customer' as const, fullName: u.displayName ?? '', email: u.email ?? '',
    phone: '', photoURL: '', fcmTokens: [] as string[], themePref: null as null | 'dark' | 'light' }
}

async function ensureUserDoc(user: User) {
  const ref = doc(db, 'r31_users', user.uid)
  const snap = await getDoc(ref)
  if (!snap.exists()) {
    await setDoc(ref, { ...newUserDoc(user), createdAt: serverTimestamp(), updatedAt: serverTimestamp() })
  }
}

export async function registerWithEmail(name: string, email: string, password: string) {
  const cred = await createUserWithEmailAndPassword(auth, email, password)
  await updateProfile(cred.user, { displayName: name })
  await ensureUserDoc(cred.user)
}
export async function loginWithEmail(email: string, password: string) {
```

```
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isSignedIn() { return request.auth != null; }
    function isOwner() { return isSignedIn() && request.auth.token.role == 'owner'; }
    function isSelf(uid) { return isSignedIn() && request.auth.uid == uid; }

    match /r31_users/{uid} {
      allow read: if isSelf(uid) || isOwner();
      allow create: if isSelf(uid) && request.resource.data.role == 'customer';
      allow update: if (isSelf(uid) && request.resource.data.role == resource.data.role) || isOwner();
      allow delete: if false;
    }
    match /r31_services/{id}   { allow read: if true; allow write: if isOwner(); }
    match /r31_phoneModels/{id}{ allow read: if true; allow write: if isOwner(); }
    match /r31_prices/{id}     { allow read: if true; allow write: if isOwner(); }
    match /r31_faqs/{id}       { allow read: if true; allow write: if isOwner(); }
    match /r31_settings/{id}   { allow read: if true; allow write: if isOwner(); }

    match /r31_orders/{id} {
      allow read: if isOwner() || (isSignedIn() && resource.data.customerId == request.auth.uid);
      allow create: if isSignedIn() && request.resource.data.customerId == request.auth.uid;
      allow update, delete: if isOwner();
      match /events/{eventId} {
        allow read: if isOwner() || (isSignedIn() && get(/databases/$(database)/documents/r31_orders/$(id)).data.customerId == request.auth.uid);
        allow create: if isOwner();
        allow update, delete: if false;
      }
    }
    match /r31_invoices/{id} {
      allow read: if isOwner() || (isSignedIn() && resource.data.customerId == request.auth.uid);
      allow write: if isOwner();
    }
    match /r31_warranties/{id} {
      allow read: if isOwner() || (isSignedIn() && resource.data.customerId == request.auth.uid);
      allow write: if isOwner();
      match /claims/{claimId} {
        allow read: if isOwner() || (isSignedIn() && get(/databases/$(database)/documents/r31_warranties/$(id)).data.customerId == request.auth.uid);
        allow create: if isSignedIn();
        allow update: if isOwner();
        allow delete: if false;
      }
    }
    match /r31_notifications/{id} {
      allow read, update: if isSignedIn() && resource.data.userId == request.auth.uid;
```
